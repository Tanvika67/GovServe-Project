Retrieve membership plan details for a specific ID.

                        Console.WriteLine("Enter membership plan id");

                        else
                        {
                            Console.WriteLine("Membership plan not found.");
                        }
                        break;
    
                    case 3:
                        Console.WriteLine("Enter membership plan id");
                        string idToUpdate = Console.ReadLine();

                            Console.WriteLine("Invalid price.");

                        if (updatedPlan != null)

                        }
                        else
                        {
                            Console.WriteLine("Membership plan not found.");
                        }
                        break;
    
                    case 4:
                        Console.WriteLine("Enter membership plan id");
                        string idToRemove = Console.ReadLine();
    
                        bool removed = repository.RemoveMembershipPlan(idToRemove);

                        {
                            Console.WriteLine("Membership plan not found.");
                        }
                        break;
    
                    case 5:
                        Console.WriteLine("Thank you");
